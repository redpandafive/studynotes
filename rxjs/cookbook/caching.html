Caching an Observable

  • Useful in a service to cut down on API calls for values that do not change
  • Use the shareReplay() operator to cache the result of the HTTP request
Example
@Injectable({
providedIn: 'root'
})
export class ApiService {
private baseUrl = 'http://example.com/api';
private cachedData: Observable<any>;

constructor(private http: HttpClient) { }

get(): Observable<any> {
if (!this.cachedData) {
this.cachedData = this.http.get(`${this.baseUrl}/data`).pipe(
shareReplay()
);
}
this.cachedData;
}
}
